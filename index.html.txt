const PROJECT_COLORS = [
  '#58a6ff','#3fb950','#f78166',
  '#d2a8ff','#ffa657','#39d353',
  '#ff7b72','#79c0ff'
];
let projects        = {};   // { id: {name, color} }
let tasks           = {};   // { projectId: [ {...frappe task} ] }
let currentProject  = null;
let ganttInstance   = null;
let editingTaskId   = null;
let selectedColor   = PROJECT_COLORS[0];
let unsubscribes    = {};

/* ════ Firebase helpers ════ */
function fbReady() { return !!window._db; }

async function loadProjects() {
  if (!fbReady()) return;
  const snap = await window._getDocs(window._collection(window._db, 'projects'));
  projects = {};
  snap.forEach(d => { projects[d.id] = d.data(); });
  renderProjectList();
}

async function loadTasks(projectId) {
  if (!fbReady()) return [];
  const snap = await window._getDocs(
    window._collection(window._db, 'projects', projectId, 'tasks')
  );
  const list = [];
  snap.forEach(d => list.push({ id: d.id, ...d.data() }));
  return list;
}

async function saveProjectToFb(id, data) {
  await window._setDoc(window._doc(window._db, 'projects', id), data);
}
async function deleteProjectFromFb(id) {
  // delete tasks first
  const snap = await window._getDocs(
    window._collection(window._db, 'projects', id, 'tasks')
  );
  for (const d of snap.docs)
    await window._deleteDoc(window._doc(window._db, 'projects', id, 'tasks', d.id));
  await window._deleteDoc(window._doc(window._db, 'projects', id));
}
async function saveTaskToFb(projectId, task) {
  const { id, ...data } = task;
  await window._setDoc(
    window._doc(window._db, 'projects', projectId, 'tasks', id), data
  );
}
async function deleteTaskFromFb(projectId, taskId) {
  await window._deleteDoc(
    window._doc(window._db, 'projects', projectId, 'tasks', taskId)
  );
}

};

/* ════ 顏色晶片 ════ */
function buildColorChips() {
  const row = document.getElementById('p-colors');
  PROJECT_COLORS.forEach((c, i) => {
    const chip = document.createElement('div');
    chip.className = 'color-chip' + (i === 0 ? ' selected' : '');
    chip.style.background = c;
    chip.onclick = () => {
      document.querySelectorAll('.color-chip').forEach(x => x.classList.remove('selected'));
      chip.classList.add('selected');
      selectedColor = c;
    };
    row.appendChild(chip);
  });
}

  list.innerHTML = '';
  ids.forEach(id => {
    const p   = projects[id];
    const div = document.createElement('div');
    div.className = 'project-item' + (id === currentProject ? ' active' : '');
    div.innerHTML = `
      <div class="dot" style="background:${p.color}"></div>
      <span class="name">${p.name}</span>
      <button class="del-btn" title="刪除專案" onclick="confirmDeleteProject(event,'${id}')">🗑</button>
    `;
    div.onclick = () => selectProject(id);
    list.appendChild(div);
  });
}

/* ════ 選擇專案 ════ */
window.selectProject = async function(id) {
  currentProject = id;
  renderProjectList();
  document.getElementById('current-project-name').textContent = projects[id].name;
  document.getElementById('current-project-name').style.color = projects[id].color;
  document.getElementById('add-task-btn').disabled = false;
  document.getElementById('stats-bar').style.display = 'flex';

  const raw = await loadTasks(id);
  tasks[id]  = raw;
  renderGantt();
  subscribeToTasks(id);
};

/* ════ 即時同步 ════ */
function subscribeToTasks(projectId) {
  if (unsubscribes[projectId]) return; // already subscribed
  const colRef = window._collection(window._db, 'projects', projectId, 'tasks');
  unsubscribes[projectId] = window._onSnapshot(colRef, snap => {
    const list = [];
    snap.forEach(d => list.push({ id: d.id, ...d.data() }));
    tasks[projectId] = list;
    if (currentProject === projectId) renderGantt();
  });
}

/* ════ 甘特圖渲染 ════ */
function renderGantt() {
  const wrap     = document.getElementById('gantt-wrap');
  const empty    = document.getElementById('empty-state');
  const taskList = tasks[currentProject] || [];

  if (!taskList.length) {
    wrap.style.display = 'none';
    empty.style.display = 'flex';
    empty.innerHTML = '<div class="icon">📋</div><div>此專案尚無任務，點右上角「＋ 新增任務」開始！</div>';
    updateStats([]);
    return;
  }

  wrap.style.display = 'block';
  empty.style.display = 'none';

  const color = projects[currentProject]?.color || 'var(--accent)';

  // 注入顏色 CSS
  const styleId = 'gantt-color-style';
  let styleEl = document.getElementById(styleId);
  if (!styleEl) { styleEl = document.createElement('style'); styleEl.id = styleId; document.head.appendChild(styleEl); }
  styleEl.textContent = `.gantt .bar { fill: ${color}88; } .gantt .bar-progress { fill: ${color}; }`;

  const frappe = taskList.map(t => ({
    id:           t.id,
    name:         t.name,
    start:        t.start,
    end:          t.end,
    progress:     t.progress || 0,
    dependencies: t.dependencies || ''
  }));

  const currentView = document.querySelector('.view-btn.active')?.textContent.trim();
  const viewMap     = { '日':'Day', '週':'Week', '月':'Month', '年':'Year' };

  try {
    ganttInstance = new Gantt('#gantt', frappe, {
      view_mode:       viewMap[currentView] || 'Week',
      date_format:     'YYYY-MM-DD',
      bar_height:      32,
      bar_corner_radius: 4,
      arrow_curve:     5,
      padding:         18,
      show_expected_progress: true,
      today_button:    true,
      language:        'zh',
      on_date_change:  onDateChange,
      on_progress_change: onProgressChange,
      on_click:        onTaskClick,
      popup: task => `
        <div class="title">${task.name}</div>
        <div class="subtitle">
          ${task.start} → ${task.end}<br/>
          進度：${task.progress}%
        </div>
        <button class="popup-button" onclick="openTaskModal('${task.id}')">✏️ 編輯</button>
      `
    });
  } catch(e) {
    console.error('Gantt render error', e);
  }

  updateStats(taskList);
}

function onDateChange(task, start, end) {
  const fmt = d => d.toISOString().split('T')[0];
  const list = tasks[currentProject];
  const t    = list.find(x => x.id === task.id);
  if (t) {
    t.start = fmt(start);
    t.end   = fmt(end);
    saveTaskToFb(currentProject, t);
    showToast('✅ 日期已更新');
  }
}

function onProgressChange(task, progress) {
  const list = tasks[currentProject];
  const t    = list.find(x => x.id === task.id);
  if (t) {
    t.progress = Math.round(progress);
    saveTaskToFb(currentProject, t);
    showToast('✅ 進度已更新');
  }
}

function onTaskClick(task) {
  // handled by popup button
}

/* ════ 統計 ════ */
function updateStats(list) {
  const total    = list.length;
  const done     = list.filter(t => t.progress >= 100).length;
  const progress = list.filter(t => t.progress > 0 && t.progress < 100).length;
  const todo     = total - done - progress;
  const avg      = total ? Math.round(list.reduce((s, t) => s + (t.progress || 0), 0) / total) : 0;
  document.getElementById('stat-total').textContent    = total;
  document.getElementById('stat-done').textContent     = done;
  document.getElementById('stat-progress').textContent = progress;
  document.getElementById('stat-todo').textContent     = todo;
  document.getElementById('stat-avg').textContent      = avg + '%';
}

/* ════ 視圖切換 ════ */
window.setView = function(mode) {
  document.querySelectorAll('.view-btn').forEach(b => b.classList.remove('active'));
  const map = { Day:'日', Week:'週', Month:'月', Year:'年' };
  document.querySelectorAll('.view-btn').forEach(b => {
    if (b.textContent.trim() === map[mode]) b.classList.add('active');
  });
  if (ganttInstance) ganttInstance.change_view_mode(mode);
};

window.confirmDeleteProject = async function(e, id) {
  e.stopPropagation();
  if (!confirm(`確定刪除專案「${projects[id].name}」？此操作無法復原。`)) return;
  if (currentProject === id) {
    currentProject = null;
    document.getElementById('current-project-name').textContent = '← 請選擇專案';
    document.getElementById('current-project-name').style.color = 'var(--text-muted)';
    document.getElementById('add-task-btn').disabled = true;
    document.getElementById('gantt-wrap').style.display = 'none';
    document.getElementById('empty-state').style.display = 'flex';
    document.getElementById('empty-state').innerHTML = '<div class="icon">📋</div><div>選擇左側專案，或新增第一個專案</div>';
    document.getElementById('stats-bar').style.display = 'none';
    ganttInstance = null;
  }
  delete projects[id];
  delete tasks[id];
  await deleteProjectFromFb(id);
  renderProjectList();
  showToast('🗑 專案已刪除');
};

/* ════ 任務 Modal ════ */
window.openTaskModal = function(taskId) {
  editingTaskId = taskId || null;
  const isEdit  = !!taskId;
  document.getElementById('task-modal-title').textContent = isEdit ? '編輯任務' : '新增任務';
  document.getElementById('t-delete-btn').style.display = isEdit ? 'block' : 'none';

  if (isEdit) {
    const t = (tasks[currentProject] || []).find(x => x.id === taskId);
    if (!t) return;
    document.getElementById('t-name').value     = t.name;
    document.getElementById('t-start').value    = t.start;
    document.getElementById('t-end').value      = t.end;
    document.getElementById('t-progress').value = t.progress || 0;
    document.getElementById('t-progress-val').textContent = (t.progress || 0) + '%';
    document.getElementById('t-deps').value     = t.dependencies || '';
  } else {
    const today = new Date().toISOString().split('T')[0];
    const next  = new Date(Date.now() + 7*86400000).toISOString().split('T')[0];
    document.getElementById('t-name').value     = '';
    document.getElementById('t-start').value    = today;
    document.getElementById('t-end').value      = next;
    document.getElementById('t-progress').value = 0;
    document.getElementById('t-progress-val').textContent = '0%';
    document.getElementById('t-deps').value     = '';
  }

  document.getElementById('task-modal').classList.add('open');
  setTimeout(() => document.getElementById('t-name').focus(), 200);
};

window.closeTaskModal = function() {
  document.getElementById('task-modal').classList.remove('open');
  editingTaskId = null;
};

window.saveTask = async function() {
  const name  = document.getElementById('t-name').value.trim();
  const start = document.getElementById('t-start').value;
  const end   = document.getElementById('t-end').value;
  const prog  = parseInt(document.getElementById('t-progress').value);
  const deps  = document.getElementById('t-deps').value.trim();

  if (!name)  { showToast('⚠️ 請輸入任務名稱'); return; }
  if (!start || !end) { showToast('⚠️ 請選擇日期'); return; }
  if (start > end)    { showToast('⚠️ 開始日期不能晚於結束日期'); return; }

  const task = {
    id:           editingTaskId || ('task-' + Date.now()),
    name, start, end,
    progress:     prog,
    dependencies: deps
  };

  if (!tasks[currentProject]) tasks[currentProject] = [];
  const idx = tasks[currentProject].findIndex(t => t.id === task.id);
  if (idx >= 0) tasks[currentProject][idx] = task;
  else          tasks[currentProject].push(task);

  await saveTaskToFb(currentProject, task);
  closeTaskModal();
  renderGantt();
  showToast(editingTaskId ? '✅ 任務已更新' : '🎉 任務已新增');
};

window.deleteTask = async function() {
  if (!editingTaskId) return;
  if (!confirm('確定刪除此任務？')) return;
  tasks[currentProject] = (tasks[currentProject] || []).filter(t => t.id !== editingTaskId);
  await deleteTaskFromFb(currentProject, editingTaskId);
  closeTaskModal();
  renderGantt();
  showToast('🗑 任務已刪除');
};

/* ════ Toast ════ */
function showToast(msg, duration = 2200) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.classList.add('show');
  setTimeout(() => el.classList.remove('show'), duration);
}
